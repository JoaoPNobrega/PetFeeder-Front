ESP32

O objetivo inicial foi entender o funcionamento do ESP32 e atender aos requisitos da disciplina de Sistemas Embarcados. Começamos com um sensor PIR, mas optamos pelo ultrassônico, que se mostrou mais preciso para o projeto.

Também testamos o uso do potenciômetro, que controla o tempo até o motor retornar à posição inicial, e adicionamos um LED para sinalizar a abertura do alimentador.

Enfrentamos alguns desafios, como fios com mau contato e um superaquecimento nas primeiras tentativas — parte natural do processo de aprendizagem e experimentação com o ESP32.

Protótipo

A ideia inicial era utilizar impressão 3D, mas, por falta de experiência, optei por uma versão caseira em papelão, focada apenas em testar a interação do sistema com o mundo real.

Apesar da simplicidade, esse protótipo trouxe muito aprendizado. Pude entender melhor como trabalhar com o motor servo e sua função de abrir a porta, além de refletir sobre o que deveria ou não ficar visível no projeto.

Ao final, decidi expor apenas o potenciômetro e o sensor ultrassônico.

Versão Final

Esta versão teve como foco corrigir pequenos problemas da estrutura de papelão, como a saída irregular da ração, causada pela forma como havia sido construída, a falta de fixação do potenciômetro e a ausência de proteção adequada para o sensor ultrassônico.

Para a construção, utilizamos o aplicativo Fusion 360 para modelagem e aplicamos corte a laser nas peças.

Essa versão recebeu o nome PetFeed AT em referência ao Amostra-Tech, a feira de ciências do CESAR School — evento para o qual ela foi especialmente desenvolvida.

Uso da LED (com PWM)

int brilhoInicial = 255;
unsigned long tempoInicioAcionamento = 0;

void acionarServoNaoBloqueante(int tempoEmMs) {
  servo1.write(90);
  analogWrite(LED_PIN, brilhoInicial);
  tempoInicioAcionamento = millis();
  tempoAcionamento = tempoInicioAcionamento + tempoEmMs;
  servoAtivo = true;
}

void atualizarServo() {
  if (servoAtivo) {
    unsigned long agora = millis();

    if (agora >= tempoAcionamento) {
      servo1.write(0);
      analogWrite(LED_PIN, 0);
      servoAtivo = false;
    } else {
      int tempoTotal = tempoAcionamento - tempoInicioAcionamento;
      int tempoDecorrido = agora - tempoInicioAcionamento;
      int brilhoAtual = map(tempoDecorrido, 0, tempoTotal, brilhoInicial, 0);
      brilhoAtual = constrain(brilhoAtual, 0, 255);
      analogWrite(LED_PIN, brilhoAtual);
    }
  }
}

Uso do sensor ultrassônico

#define TRIG_PIN 23
#define ECHO_PIN 22

float medirDistanciaCM() {
  digitalWrite(TRIG_PIN, LOW);
  delayMicroseconds(2);
  digitalWrite(TRIG_PIN, HIGH);
  delayMicroseconds(10);
  digitalWrite(TRIG_PIN, LOW);

  long duracao = pulseIn(ECHO_PIN, HIGH, 30000);
  float distancia = duracao * 0.034 / 2;
  return distancia;
}

void loop() {
  float distancia = medirDistanciaCM();

  if (distancia > 0 && distancia < 20) {
    acionarServoNaoBloqueante(tempoEmMs);
    // enviarDadosParaFirebase(distancia, potValue, tempoEmMs, origemPrioritaria, "Abertura automática por sensor");
  }
}

Uso do potenciômetro

#define POTENTIOMETER_PIN 32

int ultimoPotIndice = -1;
String origemPrioritaria = "potenciometro";
int tempoEmMs = 1000;

void loop() {
  int potValue = analogRead(POTENTIOMETER_PIN);
  int indicePot = map(potValue, 0, 4095, 0, 18);
  int tempoPotenciometro = 1000 + (indicePot * 500);

  if (indicePot == 0) {
    if (origemPrioritaria != "firebase") {
      origemPrioritaria = "firebase";
      // if (!modoAP) { Firebase.setString(fbdo, "/comando/origem", "firebase"); }
    }
  } else {
    if (indicePot != ultimoPotIndice) {
      if (origemPrioritaria != "potenciometro") {
        origemPrioritaria = "potenciometro";
        // if (!modoAP) { Firebase.setString(fbdo, "/comando/origem", "potenciometro"); }
      }
      ultimoPotIndice = indicePot;
      tempoEmMs = tempoPotenciometro;
      // if (!modoAP) { Firebase.setInt(fbdo, "/comando/tempo_remoto_ms", tempoEmMs); }
    }
  }
}

Uso do Firebase

#include <FirebaseESP32.h>

FirebaseData fbdo;
FirebaseAuth auth;
FirebaseConfig config;

void iniciarFirebase() {
  config.database_url = DATABASE_URL;
  config.signer.tokens.legacy_token = DATABASE_SECRET;
  Firebase.reconnectNetwork(true);
  fbdo.setBSSLBufferSize(4096, 1024);
  Firebase.begin(&config, &auth);
}

void enviarDadosParaFirebase(float distancia, int potValue, int tempo, const String &modo, const String &evento) {
  if (modoAP || !Firebase.ready()) return;

  Firebase.setFloat(fbdo, "/estado/distancia_cm", distancia);
  Firebase.setInt(fbdo, "/estado/potenciometro", potValue);
  Firebase.setInt(fbdo, "/estado/tempo_aberto_ms", tempo);
  Firebase.setString(fbdo, "/estado/modo", modo);
  Firebase.setString(fbdo, "/estado/evento", evento);
}

Criação do Wi-Fi próprio (Access Point)

void configurarWiFiDuplo() {
  WiFi.mode(WIFI_AP_STA);
  WiFi.begin(WIFI_SSID, WIFI_PASSWORD);
  Serial.print("Conectando à rede Wi-Fi local...");
  unsigned long tempoInicial = millis();
  while (WiFi.status() != WL_CONNECTED && millis() - tempoInicial < 10000) {
    Serial.print(".");
    delay(500);
  }
  if (WiFi.status() == WL_CONNECTED) {
    Serial.print("\nConectado! IP local: http://");
    Serial.print(WiFi.localIP());
    Serial.println("/index.html");
    modoAP = false;
  } else {
    Serial.println("\nFalha ao conectar. Iniciando AP...");
    const char* ssidAP = "PetFeeder";
    const char* senhaAP = "senha1234";
    WiFi.softAP(ssidAP, senhaAP);
    Serial.print("Access Point iniciado! IP: ");
    Serial.println(WiFi.softAPIP());
    modoAP = true;
  }
}
